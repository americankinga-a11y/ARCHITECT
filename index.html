// ───────────────────────────────────────────── // MODULE 1: INDICATOR LIBRARY // Pure functions. Stateless. Input: number[] // ───────────────────────────────────────────── export function calcEMA(data, period) { if (data.length < period) return null; const k = 2 / (period + 1); let val = data.slice(0, period).reduce((a, b) => a + b, 0) / period; for (let i = period; i < data.length; i++) val = data[i] * k + val * (1 - k); return val; } export function calcRSI(data, period = 14) { if (data.length < period + 1) return null; const slice = data.slice(-(period + 1)); let gains = 0, losses = 0; for (let i = 1; i < slice.length; i++) { const d = slice[i] - slice[i - 1]; if (d > 0) gains += d; else losses -= d; } const avgGain = gains / period, avgLoss = losses / period; if (avgLoss === 0) return 100; return 100 - (100 / (1 + avgGain / avgLoss)); } export function calcBollingerBands(data, period = 20, stdDevMult = 2) { if (data.length < period) return null; const slice = data.slice(-period); const mean = slice.reduce((a, b) => a + b, 0) / period; const variance = slice.reduce((a, b) => a + (b - mean) ** 2, 0) / period; const std = Math.sqrt(variance); return { upper: mean + stdDevMult * std, middle: mean, lower: mean - stdDevMult * std, std, // bandwidthPct: useful for volatility squeeze detection bandwidthPct: (std * 2 * stdDevMult) / mean * 100 }; } export function calcMACD(data, fast = 12, slow = 26, signal = 9) { if (data.length < slow + signal) return null; const fastEMA = calcEMA(data, fast); const slowEMA = calcEMA(data, slow); if (!fastEMA || !slowEMA) return null; const macdLine = fastEMA - slowEMA; // Build MACD line history to compute signal EMA over it const macdHistory = []; for (let i = slow; i < data.length; i++) { const slice = data.slice(0, i + 1); const f = calcEMA(slice, fast), s = calcEMA(slice, slow); if (f && s) macdHistory.push(f - s); } const signalLine = calcEMA(macdHistory, signal); return { line: macdLine, signal: signalLine, histogram: signalLine !== null ? macdLine - signalLine : null }; } // ───────────────────────────────────────────── // MODULE 2: SCORE AGGREGATOR // Returns a 0-100 sentiment score + per-indicator // directional votes and raw weighted contributions. // ───────────────────────────────────────────── const DEFAULT_WEIGHTS = { EMA: 25, RSI: 25, BB: 25, MACD: 25 }; export function computeScore(prices, weights = DEFAULT_WEIGHTS, assetType = 'CRYPTO') { if (prices.length < 30) return null; // not enough data yet const price = prices[prices.length - 1]; const e9 = calcEMA(prices, 9); const e21 = calcEMA(prices, 21); const rsi = calcRSI(prices, 14); const bb = calcBollingerBands(prices, 20, 2); const macd = calcMACD(prices, 12, 26, 9); // Each raw value is normalized to [-1, +1] // +1 = maximally bullish contribution, -1 = maximally bearish let emaRaw = 0, rsiRaw = 0, bbRaw = 0, macdRaw = 0; let emaDir = 0, rsiDir = 0, bbDir = 0, macdDir = 0; // EMA crossover: spread as fraction of price, clamped if (e9 && e21) { const spread = (e9 - e21) / e21; emaRaw = Math.max(-1, Math.min(1, spread * 200)); emaDir = emaRaw > 0.05 ? 1 : emaRaw < -0.05 ? -1 : 0; } // RSI: oversold (< 35) is bullish setup, overbought (> 65) is bearish if (rsi !== null) { rsiRaw = (rsi - 50) / 50; // maps [0,100] → [-1,+1] rsiDir = rsi < 35 ? 1 // oversold → bullish : rsi > 65 ? -1 // overbought → bearish : 0; } // Bollinger Band position: price near lower band is bullish setup if (bb) { const range = bb.upper - bb.lower; const pos = range > 0 ? (price - bb.lower) / range : 0.5; // 0=lower, 1=upper bbRaw = (pos - 0.5) * 2; // maps [0,1] → [-1,+1], lower=+1 bbDir = pos < 0.25 ? 1 // near lower band → bullish : pos > 0.75 ? -1 // near upper band → bearish : 0; } // MACD histogram: positive = bullish momentum building if (macd && macd.histogram !== null) { // Scale varies per asset class — normalise by typical histogram magnitude const scale = assetType === 'FOREX' ? 5000 : 0.05; macdRaw = Math.max(-1, Math.min(1, macd.histogram * scale)); macdDir = macdRaw > 0.05 ? 1 : macdRaw < -0.05 ? -1 : 0; } const totalWeight = weights.EMA + weights.RSI + weights.BB + weights.MACD; const weightedSum = (emaRaw * weights.EMA + rsiRaw * weights.RSI + bbRaw * weights.BB + macdRaw * weights.MACD) / totalWeight; const score = Math.max(0, Math.min(100, Math.round(50 + weightedSum * 50))); const bullCount = [emaDir, rsiDir, bbDir, macdDir].filter(d => d === 1).length; const bearCount = [emaDir, rsiDir, bbDir, macdDir].filter(d => d === -1).length; return { score, price, e9, e21, rsi, bb, macd, dirs: { ema: emaDir, rsi: rsiDir, bb: bbDir, macd: macdDir }, raws: { ema: emaRaw, rsi: rsiRaw, bb: bbRaw, macd: macdRaw }, confluence: bullCount === 4 ? 'BULL' : bearCount === 4 ? 'BEAR' : 'NONE', bullCount, bearCount }; } // ───────────────────────────────────────────── // MODULE 3: ALERT ENGINE // Fires "CALL" or "PUT" only on full confluence. // Debounced — won't re-fire until score resets. // ───────────────────────────────────────────── export class AlertEngine { constructor({ callThreshold = 90, putThreshold = 10, onAlert, onReset } = {}) { this.callThreshold = callThreshold; this.putThreshold = putThreshold; this.onAlert = onAlert || (() => {}); this.onReset = onReset || (() => {}); this._lastFired = null; // 'CALL' | 'PUT' | null this._alertCount = 0; } evaluate(result) { if (!result) return; const { score, confluence } = result; const resetZone = score > this.putThreshold + 10 && score < this.callThreshold - 10; // Reset latch when score returns to neutral zone if (resetZone && this._lastFired !== null) { this._lastFired = null; this.onReset({ score }); } // CALL: score above threshold AND all four indicators bullish if (score >= this.callThreshold && confluence === 'BULL' && this._lastFired !== 'CALL') { this._alertCount++; this._lastFired = 'CALL'; this.onAlert({ type: 'CALL', score, confluence, alertId: this._alertCount, timestamp: new Date().toISOString(), indicators: result.dirs }); } // PUT: score below threshold AND all four indicators bearish if (score <= this.putThreshold && confluence === 'BEAR' && this._lastFired !== 'PUT') { this._alertCount++; this._lastFired = 'PUT'; this.onAlert({ type: 'PUT', score, confluence, alertId: this._alertCount, timestamp: new Date().toISOString(), indicators: result.dirs }); } } setThresholds(call, put) { this.callThreshold = call; this.putThreshold = put; } } // ───────────────────────────────────────────── // MODULE 4: CANDLE BUFFER // Rolling window. Feed raw ticks or OHLCV candles. // ───────────────────────────────────────────── export class CandleBuffer { constructor(maxLength = 300) { this.closes = []; this.maxLength = maxLength; } // Feed a close price directly (from tick data) pushClose(close) { this.closes.push(close); if (this.closes.length > this.maxLength) this.closes.shift(); } // Feed a full OHLCV candle object { open, high, low, close, volume } pushCandle(candle) { this.pushClose(candle.close); } get() { return this.closes; } get length() { return this.closes.length; } clear() { this.closes = []; } } // ───────────────────────────────────────────── // MODULE 5: BINANCE WEBSOCKET ADAPTER // Drop-in feed for live data. Swap URL for // other providers (OANDA uses similar WS format). // ───────────────────────────────────────────── export class BinanceFeed { constructor({ symbol = 'btcusdt', interval = '1m', onCandle, onError } = {}) { this.symbol = symbol.toLowerCase(); this.interval = interval; this.onCandle = onCandle || (() => {}); this.onError = onError || console.error; this.ws = null; this._reconnectDelay = 2000; } connect() { const url = `wss://stream.binance.com:9443/ws/${this.symbol}@kline_${this.interval}`; this.ws = new WebSocket(url); this.ws.onopen = () => console.log(`[BinanceFeed] Connected: ${this.symbol} ${this.interval}`); this.ws.onmessage = (evt) => this._handleMessage(evt); this.ws.onerror = (err) => this.onError(err); this.ws.onclose = () => { console.warn('[BinanceFeed] Disconnected. Reconnecting...'); setTimeout(() => this.connect(), this._reconnectDelay); }; } _handleMessage(evt) { try { const msg = JSON.parse(evt.data); const k = msg.k; // kline object // Only emit on candle CLOSE to avoid repainting if (!k.x) return; // k.x = isFinal (candle closed) this.onCandle({ open: parseFloat(k.o), high: parseFloat(k.h), low: parseFloat(k.l), close: parseFloat(k.c), volume: parseFloat(k.v), openTime: k.t, closeTime: k.T }); } catch (e) { this.onError(e); } } disconnect() { if (this.ws) { this.ws.onclose = null; this.ws.close(); this.ws = null; } } } // ───────────────────────────────────────────── // WIRING EXAMPLE: connect all modules together // ───────────────────────────────────────────── /* import { CandleBuffer, BinanceFeed, computeScore, AlertEngine } from './signalEngine.js'; const buffer = new CandleBuffer(300); const alerts = new AlertEngine({ callThreshold: 90, putThreshold: 10, onAlert: (alert) => { // Your UI handler — update the dashboard here console.log(`[ALERT] ${alert.type} | score=${alert.score} | id=${alert.alertId}`); dispatchEvent(new CustomEvent('signal-alert', { detail: alert })); }, onReset: ({ score }) => { console.log(`[RESET] Score returned to neutral: ${score}`); } }); const feed = new BinanceFeed({ symbol: 'btcusdt', interval: '1m', onCandle: (candle) => { buffer.pushCandle(candle); const result = computeScore(buffer.get(), { EMA: 25, RSI: 25, BB: 25, MACD: 25 }, 'CRYPTO'); if (result) alerts.evaluate(result); } }); feed.connect(); कैसा प्रोग्राम */
